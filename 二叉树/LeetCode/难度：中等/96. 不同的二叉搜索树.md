class Solution {
    public int numTrees(int n) {
        int[] G = new int[n+1];
        G[0] = 1;
        G[1] = 1;
        for(int i = 2; i <= n; i++){
            for(int j = 1 ;j <=i; j++){
                G[i] += G[j-1] * G[i-j]; 
            }
        }
        return G[n];
    }
}

```java
class Solution {
    public int numTrees(int n) {
        if(n == 0 || n == 1){
            return 1;
        }
        return helper(1,n);
    }
    private int helper(int start, int end){
        if(start >= end){
            return 1;
        }
        int ans = 0;
        for(int i = start; i <= end; i++){
            int left = helper(start,i-1);
            int right = helper(i + 1, end);
            ans += left * right;
        }
        return ans;
    }
}
```

class Solution {
    public int numTrees(int n) {
        if(n == 0 || n == 1){
            return 1;
        }
        HashMap<Integer,Integer> map = new HashMap<>();
        return helper(n,map);
    }
    private int helper(int n, HashMap<Integer,Integer> map){
        if(map.containsKey(n)){
            return map.get(n);
        }
        if(n == 0 || n == 1){
            return 1;
        }

        int ans = 0;
        for(int i = 1; i <= n; i++){
            int left = helper(i - 1, map);
            int right = helper(n - i, map);
            ans += left * right; 
        }
        map.put(n,ans);
        return ans;
    }
}
