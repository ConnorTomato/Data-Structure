#### [面试题 04.10. 检查子树](https://leetcode-cn.com/problems/check-subtree-lcci/)

```java
class Solution {
    public boolean checkSubTree(TreeNode t1, TreeNode t2) {
        if(t2 == null){
            return true;
        }
        if(t1 == null){
            return false;
        }
        return isSubTree(t1,t2) || checkSubTree(t1.left,t2) || checkSubTree(t1.right,t2);
    }
    private boolean isSubTree(TreeNode t1, TreeNode t2){
        if(t2 == null){
            return true;
        }
        if(t1 == null){
            return false;
        }
        if(t1.val != t2.val){
            return false;
        }
        return isSubTree(t1.left,t2.left) || isSubTree(t1.right,t2.left) || isSubTree(t1.right,t2.left) || isSubTree(t1.right,t2.right);
    }
}
```
