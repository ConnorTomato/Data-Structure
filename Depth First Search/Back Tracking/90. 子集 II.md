#### [90. 子集 II](https://leetcode-cn.com/problems/subsets-ii/)

> - 执行用时：6 ms, 在所有 Java 提交中击败了8.64%的用户

```java
class Solution {
    public List<List<Integer>> subsetsWithDup(int[] nums) {
        Arrays.sort(nums);
        HashSet<List<Integer>> res = new HashSet<List<Integer>>();
        List<Integer> path = new LinkedList<Integer>();
        backtracking(nums, 0, res, path);
        return new LinkedList<>(res);        
    }
    private void backtracking(int[] nums, int index, HashSet<List<Integer>> res, List<Integer> path){
        if(index == nums.length){
            res.add(new LinkedList<>(path));
            return;
        }
        path.add(nums[index]);
        backtracking(nums, index + 1, res, path);

        path.remove(path.size() - 1);
        backtracking(nums, index + 1, res, path);
    }
}
```
