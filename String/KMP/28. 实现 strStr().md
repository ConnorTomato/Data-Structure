#### [28. 实现 strStr()](https://leetcode-cn.com/problems/implement-strstr/)

> - 执行用时：3 ms, 在所有 Java 提交中击败了39.16%的用户

```c
class Solution {
    public int strStr(String haystack, String needle) {
        char[] t = haystack.toCharArray();
        char[] p = needle.toCharArray();
        int[] next = new int[p.length + 1];
        getNext(p, next);
        return KMP(t, p , next);
    }
    private int KMP(char[] t, char[] p, int[] next){
        int i = 0, j = 0;
        while(i < t.length && j < p.length){
            if(j == -1 || t[i] == p[j]){
                i++;
                j++;
            }else{
                j = next[j];
            }
        }
        if(j == p.length){
            return i - j;
        }
        return -1;
    }
    private void getNext(char[] p, int[] next){
        next[0] = -1;
        int i = 0, j = -1;
        while(i < p.length){
            if(j == -1 || p[i] == p[j]){
                i++;
                j++;
                next[i] = j;
            }else{
                j = next[j];
            }
        }
    }
}
```
